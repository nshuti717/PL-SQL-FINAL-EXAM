### SQL Query for Dashboard Data

```sql
-- Executive Dashboard KPIs
SELECT 
    (SELECT COUNT(*) FROM mothers) AS total_mothers,
    (SELECT COUNT(*) FROM pregnancies WHERE status = 'ACTIVE') AS active_pregnancies,
    (SELECT COUNT(*) FROM pregnancies WHERE status = 'ACTIVE' AND risk_level = 'HIGH') AS high_risk,
    (SELECT COUNT(*) FROM maternal_alerts WHERE severity = 'RED' AND status = 'OPEN') AS red_alerts,
    (SELECT COUNT(*) FROM maternal_alerts WHERE severity = 'ORANGE' AND status = 'OPEN') AS orange_alerts,
    (SELECT COUNT(*) FROM maternal_alerts WHERE severity = 'YELLOW' AND status = 'OPEN') AS yellow_alerts
FROM dual;
```

### SQL Query for Audit Dashboard

```sql
-- Audit Dashboard Summary
SELECT 
    (SELECT COUNT(*) FROM audit_log) AS total_operations,
    (SELECT COUNT(*) FROM audit_log WHERE status = 'ALLOWED' OR status IS NULL) AS allowed,
    (SELECT COUNT(*) FROM audit_log WHERE status = 'DENIED') AS denied,
    (SELECT COUNT(*) FROM audit_log WHERE TRUNC(change_timestamp) = TRUNC(SYSDATE)) AS today_ops
FROM dual;

-- Denied by day of week
SELECT 
    TO_CHAR(change_timestamp, 'DY') AS day_of_week,
    COUNT(*) AS denied_count
FROM audit_log
WHERE status = 'DENIED'
GROUP BY TO_CHAR(change_timestamp, 'DY')
ORDER BY DECODE(TO_CHAR(change_timestamp, 'DY'), 'MON', 1, 'TUE', 2, 'WED', 3, 'THU', 4, 'FRI', 5, 'SAT', 6, 'SUN', 7);
```

### SQL Query for CHW Dashboard

```sql
-- CHW Performance Summary
SELECT 
    c.chw_id,
    c.first_name || ' ' || c.last_name AS chw_name,
    c.sector,
    COUNT(DISTINCT p.pregnancy_id) AS active_cases,
    COUNT(DISTINCT CASE WHEN a.status = 'OPEN' THEN a.alert_id END) AS open_alerts,
    ROUND(AVG(CASE WHEN sch.status = 'COMPLETED' THEN 100 ELSE 0 END), 0) AS anc_completion_pct
FROM community_health_workers c
LEFT JOIN pregnancies p ON c.chw_id = p.assigned_chw_id AND p.status = 'ACTIVE'
LEFT JOIN maternal_alerts a ON p.pregnancy_id = a.pregnancy_id
LEFT JOIN anc_schedule sch ON p.pregnancy_id = sch.pregnancy_id
WHERE c.status = 'ACTIVE'
GROUP BY c.chw_id, c.first_name, c.last_name, c.sector
ORDER BY active_cases DESC;
```

### SQL Query for Alert Dashboard

```sql
-- Open Alerts by Severity
SELECT 
    a.alert_id,
    a.severity,
    a.risk_score,
    m.first_name || ' ' || m.last_name AS mother_name,
    m.phone,
    m.sector,
    c.first_name || ' ' || c.last_name AS assigned_chw,
    a.alert_message,
    a.created_at
FROM maternal_alerts a
JOIN pregnancies p ON a.pregnancy_id = p.pregnancy_id
JOIN mothers m ON p.mother_id = m.mother_id
JOIN community_health_workers c ON p.assigned_chw_id = c.chw_id
WHERE a.status = 'OPEN'
ORDER BY DECODE(a.severity, 'RED', 1, 'ORANGE', 2, 'YELLOW', 3), a.risk_score DESC;

-- Ambulance Status
SELECT vehicle_number, driver_name, status FROM ambulances ORDER BY status;
```
